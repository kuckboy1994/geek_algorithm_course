1. 方法名称

* 时间复杂度: O(nlogn)
* 空间复杂度: O(n)

```js
var topKFrequent = function (nums, k) {
  const hash = new Map();
  for (let i = 0; i < nums.length; i++) {
    hash.set(nums[i], (hash.get(nums[i]) || 0) + 1);
  }

  const t = [...hash.entries()].sort((a, b) => b[1] - a[1]);
  const res = [];
  for (let i = 0; i < k; i++) {
    res.push(t[i][0]);
  }

  return res;
};
```
