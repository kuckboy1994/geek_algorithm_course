# 69. x 的平方根

1. 二分法

* 时间复杂度: O(logn)
* 空间复杂度: O(1)

```js
/**
 * @param {number} x
 * @return {number}
 */
var mySqrt = function (x) {
  if (x == 0 || x == 1) {
    return x;
  }
  let left = 1;
  let right = x;
  let mid = 1;

  while (left <= right) {
    mid = left + ((right - left) >> 1);
    // console.log(left, right, mid);
    if (mid * mid > x) {
      right = mid - 1;
    } else {
      left = mid + 1;
    }
  }

  return right;
};
```
